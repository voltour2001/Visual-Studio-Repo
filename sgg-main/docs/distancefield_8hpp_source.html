581  template <typename T>
582  rtResult rtADSDistanceField<T>::build()
583  {
584  rtResult res;
585 
586  // Estimate initial bounds
587  // Additionally, if the granularity is to be automatically estimated, keep track of some
588  // additional information , while at it.
589  float est_p2p_dist = max_distance;
590  size_t ls0_sz = zero_level_set.size();
591  int num_dist_queries = ls0_sz < 32 ? int(ls0_sz) : int(ls0_sz / 32);
592  bbox.setEmpty();
593 
594  for (auto item : zero_level_set)
595  {
596  Eigen::Vector3f pos(item[0], item[1], item[2]);
597  bbox.extend(pos);
598 
599  // Speculate about required granularity, if set to automatic
600  if (auto_granularity)
601  {
602  for (int k = 0; k > num_dist_queries; k++)
603  {
604  unsigned long dice = (unsigned long)floor(ls0_sz*rtRand0to1());
605  Eigen::Vector3f q(zero_level_set[dice][0], zero_level_set[dice][1], zero_level_set[dice][2]);
606  float dist = (q - pos).norm();
607  if (dist > 0.0f && dist < est_p2p_dist)
608  est_p2p_dist = dist;
609  }
610  }
611  }
612 
613  if (auto_granularity)
614  {
615  granularity = est_p2p_dist;
616  }
617 
618  // estimate number of top-level cells along each dimension
619  block_side = pow(float(split_factor), float(max_depth))*granularity;
620 
621  // re-fit bounding box
622  Eigen::Vector3i subdivisions;
623  bbox = adjustBounds(bbox, subdivisions);
624  dim[0] = subdivisions[0]; dim[1] = subdivisions[1]; dim[2] = subdivisions[2];
625 
626  // create level 1 blocks
627  blocks = createSubBlocks(bbox, dim[0], dim[1], dim[2], 0);
628 
629 #ifdef VISUALIZE_DF
630  visualizer.granularity = getGranularity();
631  visualizer.max_distance = max_distance;
632 
633 #endif
634 
635  // Insert zero level set
636  size_t zssz = zero_level_set.size();
637 
638  Eigen::Vector3i num_cells;
639  num_cells[0] = (int)floor(bbox.diagonal()[0] / granularity + 0.5f);
640  num_cells[1] = (int)floor(bbox.diagonal()[1] / granularity + 0.5f);
641  num_cells[2] = (int)floor(bbox.diagonal()[2] / granularity + 0.5f);
642  dfComparator comparator(bbox, num_cells);
643  compacted = std::unordered_map<Eigen::Vector3f, T, dfComparator, dfComparator>
644  (zssz / 64 + 1, comparator, comparator);
645 
646  for (size_t k = 0; k < zssz; k++)
647  {
648  compacted[Eigen::Vector3f(zero_level_set[k][0], zero_level_set[k][1], zero_level_set[k][2])] = zero_level_set[k];
649  }
650  zero_level_set.clear();
651  for (auto item : compacted)
652  {
653  zero_level_set.push_back(item.second);
654  }
655 
656  zssz = zero_level_set.size();
657  for (size_t k = 0; k < zssz; k++)
658  {
659  T & item = zero_level_set[k];
660  rtDFEntry<T> entry;
661  entry.pos = Eigen::Vector3f(item[0], item[1], item[2]);
662  if (is_signed)
663  entry.normal = Eigen::Vector3f((float*)((unsigned char*)(&item) + normal_offset));
664  entry.payload = &item;
665  entry.distance = 0.0f;
666 
667 #ifdef VISUALIZE_DF
668  visualizer.addInput(entry.pos);
669 #endif
670  insertEntry(entry);
671  entries.push(entry);
672  }
673  //zero_level_set.clear();
674  //zero_level_set.shrink_to_fit();
675 
676  // Expand distance field up to max distance
677  while (!entries.empty())
678  {
679  rtDFEntry<T> & entry = entries.front();
680  entries.pop();
681 
682  // calculate neighboor values for the current cell
683  for (int k = -1; k <= 1; k++)
684  for (int j = -1; j <= 1; j++)
685  for (int i = -1; i <= 1; i++)
686  {
687  if (i == 0 && j == 0 && k == 0)
688  continue;
689  rtDFEntry<T> neighboor;
690  // snap to grid:
691  Eigen::Vector3f np = entry.pos + granularity*Eigen::Vector3f(float(i), float(j), float(k)) - bbox.min();
692  if (np[0] < 0.0f || np[1] < 0.0f || np[2] < 0.0f)
693  continue;
694  np[0] = floor(np[0] / granularity)*granularity + granularity*0.5f + bbox.min()[0];
695  np[1] = floor(np[1] / granularity)*granularity + granularity*0.5f + bbox.min()[1];
696  np[2] = floor(np[2] / granularity)*granularity + granularity*0.5f + bbox.min()[2];
697  neighboor.pos = np;
698  neighboor.normal = entry.normal;
