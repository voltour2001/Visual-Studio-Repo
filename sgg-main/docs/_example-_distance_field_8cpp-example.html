Example showing how to genere a signed distance field from a mesh.

void DistanceFieldExample()
{
// Load test model
// We will load normal vectors along with the positional data to make a directed (signed) distance field.
// For sample storage we will use an std::array<float,6>
//
rt::rtPLYReader<std::array<float, 6>> reader(rt::attribute_t::RT_ATTRIB_POSITION | rt::attribute_t::RT_ATTRIB_NORMAL);
rt::rtResult res = reader.setAttributeOffset(rt::attribute_t::RT_ATTRIB_NORMAL, 3 * sizeof(float));
std::vector<std::array<float, 6>> vertices;
std::vector<long> indices;
std::string in_file = "DoraBlock2_bin.ply";
std::ifstream in_stream;
res = rt::rtOpenFileStream(&in_stream, in_file);
if (res)
res = reader.readMesh(&in_stream, &vertices, &indices);
else
{
std::cout << res.message << std::endl;
return;
}
if (!res)
{
std::cout << res.message << std::endl;
return;
}
// Sample the surface do optain a zero level set to inject to the distance field.
// To do so, instanciate a mesh scan converter, configure it and generate the surface samples:
//
std::vector<std::array<float, 6>> samples;
rt::rtScanConverter<std::array<float, 6>> rasterizer;
rasterizer.setNumAttribCoefs(3);
rasterizer.useAttributeInterpolation(true);
rasterizer.setStride(4.0f);
rasterizer.useVoxelCenters(true);
rasterizer.setSampleProbability(1.0f);
rasterizer.setSampleJitter(0.0f);
size_t num_samples = rasterizer.convertMesh(samples, vertices, indices);
// Create an empty distance field using a granularity of 5.0 units, a truncation distance of 60.0,
// depth 3 (number of non-leaf layers) and a split factor of 4 per cell side (64 cells per subdivision)
rt::rtDistanceField<std::array<float, 6>> df(5.0f,60.0f,3,4);
// Enable distance sign and provide the offset in bytes in the sample data where the normal vector is located
df.setSigned(true, sizeof(float) * 3);
// Add the zero-level set samples to the distance field
df.addData(samples.data(),samples.size());
// Build the data structure. This includes generating all necessary entries up to the truncation distance.
df.build();
// The distance field is now ready to perform queries.
}
