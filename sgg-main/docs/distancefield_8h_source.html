103  std::size_t operator()(const Eigen::Vector3f & key) const
104  {
105  Eigen::Vector3f np = key - bbox.min();
106  int x, y, z;
107  x = (int)floor(np[0] / granularity);
108  y = (int)floor(np[1] / granularity);
109  z = (int)floor(np[2] / granularity);
110  return x + y*dims[0] + z*dims[0] * dims[1];
111  }
