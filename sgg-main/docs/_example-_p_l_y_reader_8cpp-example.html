Example-PLYReader.cpp

Example of using the PLYReader to import a point cloud with vertex colors.

void readPLYVertexData(std::string filename)
{
// Declare a vector to hold the vertex data. Each vertex structure consists of 3 position coordinates and 3 color components
std::vector<std::array<float, 6>> in_vertices;
// Open a stream for data input
std::ifstream str;
rt::rtResult res = rt::rtOpenFileStream(&str, filename);
// Create a reader compatible with the vertex data
rt::rtPLYReader<std::array<float, 6>> reader;
// Specify what attributes to read from the file. All attributes that are incompatible with
// the format are ignored.
setActiveAttributes((rt::attribute_t::RT_ATTRIB_POSITION | rt::attribute_t::RT_ATTRIB_COLOR);
if (res.code != rt::rtResult::RT_OK)
{
std::cout << "Could not open file for reading vertex data\n" << std::endl;
return;
}
else
{
// Initialize reader and read file header
res = reader.initialize(&str);
std::cout << res.message << std::endl;
// Specify the offset in bytes from the begining of the vertex structure where the color components begin.
// All components of the same attribute are assumed consecutive in the vertex structure.
// The position attributes use their default offset, which is 0.
res = reader.setAttributeOffset(rt::attribute_t::RT_ATTRIB_COLOR, 3 * sizeof(float));
std::cout << res.message << std::endl;
// Read as many vertices as possible, up to 100000. This is useful if there are memory
// limitations to be respected.
res = reader.readVertexData(&in_vertices, 100000);
std::cout << res.message << std::endl;
printf("Data read:\n");
for (auto s : in_vertices)
{
printf("position: %f %f %f, color: %f %f %f\n", s[0], s[1], s[2], s[3], s[4], s[5]);
}
printf("\n");
}
}
